#include <iostream>
using namespace std;

int main()
{
    double a = 1.3; // задаем значение а
    double sum = 0; // переменная для хранения суммы чисел последовательности
    
    // проходим по последовательности с помощью цикла for
    for (int n = 1; n <= 10; n++) {
        sum += 1.0 / n; // добавляем очередное число в сумму
        if (sum > a) { // если сумма больше а, выводим n и завершаем цикл
            cout << n << endl;
            break;
        }
    }
    return 0;
}
