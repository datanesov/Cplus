#include <iostream>
using namespace std;

int main()
{
    double a = 1.3; // задаем значение а
    
    // проходим по последовательности с помощью цикла for
    for (int n = 2; n <= 10; n++) {
        double num = 1 + 1.0 / n; // вычисляем текущее число последовательности
        if (num < a) break; // если число меньше а, останавливаем цикл
        else if (n == 10) cout << n << endl; // если дошли до конца последовательности и все числа больше а, выводим n
    }
    return 0;
}
