#include <iostream>
using namespace std;

int main()
{
    double n = 1; // задаем начальное значение для числа последовательности
    double sum = 1; // переменная для хранения суммы чисел последовательности
    
    // проходим по последовательности с помощью цикла while
    while (sum <= 10) {
        // выводим первое число, большее n, и завершаем цикл
        if (n < 2 && 1 + n > 2) {
            cout << 1 + n << endl;
            break;
        }
        n += 1.0 / (int)n; // вычисляем очередное число и добавляем его в сумму
        sum += 1 + n; // обновляем значение суммы
    }
    return 0;
}
