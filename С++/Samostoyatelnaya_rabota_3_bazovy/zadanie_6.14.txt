#include <iostream>
using namespace std;

int main()
{
    double a = 1.3; // задаем значение а
    double sum = 0; // переменная для хранения суммы чисел последовательности
    int n = 2; // счетчик чисел последовательности
    
    // проходим по последовательности с помощью цикла while
    while (1) {
        sum += 1.0 / n; // добавляем очередное число в сумму
        n++; // увеличиваем счетчик чисел последовательности
        
        if (1 + sum < a) continue; // если последнее число меньше а, продолжаем цикл
        else { // если последнее число больше или равно а, выводим n и завершаем цикл
            cout << n - 1 << endl;
            break;
        }
    }
    return 0;
}
