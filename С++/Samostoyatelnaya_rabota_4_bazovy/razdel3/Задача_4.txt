#include <iostream>
using namespace std;

int main() {
    // создаем квадратную матрицу 5x5 и заполняем случайными числами
    double arr[5][5] = { {4.5, 2.1, 7.8, -1.3, 8.6}, {6.4, 9.0, 3.2, 4.1, -2.9}, {5.7, 0.8, -2.1, 1.2, 3.5},
                      {-2.8, 9.3, 6.1, 5.5, 1.4}, {5.2, 7.9, 3.0, 0.7, 1.8} };
    double min = arr[4][0];
    // находим минимальный элемент на побочной диагонали
    for(int i = 0; i < 5; i++){
        if(arr[i][4-i] < min){
            min = arr[i][4-i];
        }
    }
    // выводим минимальный элемент
    cout << "Наименьший элемент на побочной диагонали: ";
    cout << min << endl;
    return 0;
}
