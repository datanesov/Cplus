
#include <iostream>
#include <cmath>
using namespace std;

int main() {
    // создаем массив 7x7 и заполняем случайными числами
    int arr[7][7] = { {4, -2, 7, 1, -8, 6, 9}, {6, 9, 3, -4, 2, -1, 8}, {5, 0, -2, 1, 3, -5, 7}, 
                      {2, 9, -6, 5, 1, 4, -3}, {-5, 7, 3, 0, -1, 8, -2}, {1, -3, 5, -2, 4, 9, 6}, 
                      {8, 2, 4, -7, 1, -9, 3} };
    int sum = 0;
    for(int i = 0; i < 7; i++){
        for(int j = 0; j < 7; j++){
            if(arr[i][j] < 0 && abs(arr[i][j]) % 2 == 1){
                sum += abs(arr[i][j]);
            }
        }
    }
    // выводим сумму
    cout << "Сумма модулей отрицательных нечетных элементов: " << sum << endl;
    return 0;
}
