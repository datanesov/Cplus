#include <iostream>
using namespace std;

int main()
{
    double v1, v2, s, t;
    cout << "Введите скорость первого автомобиля (км/ч): ";
    cin >> v1;
    cout << "Введите скорость второго автомобиля (км/ч): ";
    cin >> v2;
    cout << "Введите расстояние между автомобилями (км): ";
    cin >> s;
    cout << "Введите время (ч): ";
    cin >> t;

    double d = abs(s - (v1+v2)*t); // расстояние между автомобилями через t часов
    cout << "Расстояние между автомобилями через " << t << " часов: " << d << " км" << endl;

    return 0;
}
