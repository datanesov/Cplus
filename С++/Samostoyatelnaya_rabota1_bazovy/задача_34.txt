#include <iostream>
using namespace std;

int main() {
    float a, b;
    cout << "Введите два числа: ";
    cin >> a >> b;
    cout << "Сумма: " << a + b << endl;
    cout << "Разность: " << a - b << endl;
    cout << "Произведение: " << a * b << endl;
    cout << "Частное: " << a / b << endl;
    return 0;
}
